61 RTreeContext::vertex interpolate( const RTreeContext::vertex& qA,
62  const RTreeContext::vertex& qB,
63  double t ){
64 
65  RTreeContext::vertex qt;
66  boost::fill(qt,0.0);
67  for( std::size_t i=0; i<qt.size(); i++ )
68  { qt[i] = ( 1.0 - t )*qA[i] + t*qB[i]; }
69  return qt;
70 
71 }

175 template<typename Container>
176 void RTreeContext::make_vertices(Container& nRandomPos, std::size_t N) {
177 
178  boost::random::mt19937 rng; // produces randomness out of thin air
179  // see pseudo-random number generators
180  boost::random::uniform_real_distribution<> ptp(-boost::math::constants::pi<double>(),boost::math::constants::pi<double>());
181  // distribution that maps -pi to pi
182  // see random number distributions
183  rng.seed(static_cast<unsigned int>(std::time(0)));
184 
185  plan::ArmPos nextpos;
186 
187 
188  // create a robot state
189  moveit::core::RobotState robotstate( robotmodel );
190 
191  for(std::size_t n = 0; n < N; ++n){
192  for(std::size_t i = 0; i < plan::ArmPos::size(); ++i){
193  nextpos[i] = ptp(rng); // simulate rolling a die
194  }
195 
196  robotstate.setJointGroupPositions( "manipulator", nextpos.begin() );
197 
198  if(! getPlanningScene()->isStateColliding( robotstate, "manipulator", false ) )
199  {
200  nRandomPos.push_back(nextpos);
201  }
202 
203  }
204 }

316  boost::add_edge(
317  aposToVertex1.second,
318  aposToVertex2.second,

322  }
323  }
324
